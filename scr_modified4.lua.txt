-- ========== CƠ CHẾ LOAD ORIONLIB DỰ PHÒNG ==========
local function LoadOrion()
    local OrionLib = nil
    local urls = {
        "https://raw.githubusercontent.com/x7ry/Library/main/OrionLib.lua",
        "https://raw.githubusercontent.com/shlexware/Orion/main/source",
        "https://raw.githubusercontent.com/GreenDeno/Garden/refs/heads/main/OrionLib.lua",
        "https://raw.githubusercontent.com/MinhDuc19092007/TCP-DOWN/refs/heads/main/Orion_Aqua_Transparent.lua"
    }
    for _, url in ipairs(urls) do
        local suc, res = pcall(function()
            return loadstring(game:HttpGet(url))()
        end)
        if suc and res then
            OrionLib = res
            break
        end
    end
    if not OrionLib then
        error("Không thể tải OrionLib từ bất kỳ link nào. Vui lòng kiểm tra kết nối.")
    end
    return OrionLib
end

    local mobFolder = world:FindFirstChild("Mobs")

            if v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                return v
            end
        end
    end

    -- NẾU KHÔNG CÓ BOSS → FARM MOB THƯỜNG

        for _, v in pairs(mobFolder:GetChildren()) do
            if v:FindFirstChild("HumanoidRootPart") and v:FindFirstChild("Humanoid") and v.Humanoid.Health > 0 then
                return v
            end
        end
    end

    return nil
end

-- Hàm lấy CFrame động theo người chơi
local function GetPlayerCFrame()
    local char = game.Players.LocalPlayer.Character
    if char and char:FindFirstChild("HumanoidRootPart") then
        return char.HumanoidRootPart.CFrame
    end
    return CFrame.new(0, 100, 0) -- fallback
end

-- Tạo args cho skill
local function CreateArgs(skillId, began)
    local rootCFrame = GetPlayerCFrame()
    
    -- Điều chỉnh offset tùy game của bạn (dùng giá trị gốc bạn cung cấp làm base)
    local cameraCf = rootCFrame * CFrame.new(3.6, 2.8, 11.7)   -- approx từ ví dụ của bạn
    local playerCf  = rootCFrame * CFrame.new(-3.6, -2.8, 11.7)
    
    -- Aim: tạm hướng về phía trước, bạn có thể chỉnh thành lock enemy nếu cần
    local aimPos = rootCFrame.Position + rootCFrame.LookVector * 40
    
    return {
        {
            Camera = cameraCf,
            SkillId = skillId,
            Began = began,
            CFrame = playerCf,
            ["Typ\208\181"] = 1,
            Aim = Vector3.new(aimPos.X, aimPos.Y, aimPos.Z)
        }
    }
end

-- Hàm spam skill (chạy trong spawn)
local function StartAuto()
    local skillId = SkillIds[SelectedMode]
    if not skillId then return end
    
    spawn(function()
        while AutoEnabled do
            local args = CreateArgs(skillId, true)
            pcall(function()
                game:GetService("ReplicatedStorage").Remotes.SkillRemote:FireServer(unpack(args))
            end)
            wait(0.05)
        end
        
        -- Khi tắt auto → gửi Began = false
        local argsOff = CreateArgs(skillId, false)
        pcall(function()
            game:GetService("ReplicatedStorage").Remotes.SkillRemote:FireServer(unpack(argsOff))
        end)
    end)
end

local Section = Tab2:AddSection({
        Name = "Check time boss"
})

local bossLabels = {}

local function createLabel(name, text)
        if bossLabels[name] then return end
        bossLabels[name] = Section:AddLabel(name.." : "..text)
end

local function updateLabel(name, text)
        if bossLabels[name] then
                bossLabels[name]:Set(name.." : "..text)
        end
end

local function removeLabel(name)
        if bossLabels[name] then
                bossLabels[name]:Destroy()
                bossLabels[name] = nil
        end
end

task.spawn(function()
        while true do
                pcall(function()
                        local misc = workspace:FindFirstChild("Misc")
                        local mobFolder = misc and misc:FindFirstChild("MobSpawnInfo")

                        if mobFolder then
                                for _,mob in pairs(mobFolder:GetChildren()) do
                                        local timer = mob:FindFirstChild("MobRespawnTimer")

                                        if timer and timer:FindFirstChild("TextLabel") then
                                                local text = timer.TextLabel.Text

                                                if text ~= "00" then
                                                        if not bossLabels[mob.Name] then
                                                                createLabel(mob.Name, text)
                                                        else
                                                                updateLabel(mob.Name, text)
                                                        end
                                                else
                                                        removeLabel(mob.Name)
                                                end
                                        end
                                end
                        end
                end)

                task.wait(1)
        end
end)

-- Kiểm tra xem có timer đếm ngược spawn mob không
local function IsRespawning()
    local misc = workspace:FindFirstChild("Misc")
    local mobSpawnInfo = misc and misc:FindFirstChild("MobSpawnInfo")
    if mobSpawnInfo then
        for _, mob in pairs(mobSpawnInfo:GetChildren()) do
            local timer = mob:FindFirstChild("MobRespawnTimer")
            if timer and timer:FindFirstChild("TextLabel") then
                local text = timer.TextLabel.Text
                if text ~= "00" and text ~= "" then
                    return true -- đang đếm ngược
                end
            end
        end
    end
    return false
end

local AutoGodmode = false
local mainConnection

local HEIGHT_ABOVE = 8.8

-- Hàm lấy danh sách mục tiêu (từ Event Mobs) còn sống và sắp xếp theo máu tăng dần (ít máu nhất lên đầu)
local function GetSortedTargets()
    local targets = {}
    local eventMobs = workspace:FindFirstChild("World Mobs") and workspace["World Mobs"]:FindFirstChild("Event Mobs")
    if not eventMobs then return targets end
    
    for _, mob in pairs(eventMobs:GetChildren()) do
        local humanoid = mob:FindFirstChild("Humanoid")
        local root = mob:FindFirstChild("HumanoidRootPart")
        -- Chỉ lấy những con còn sống và có HumanoidRootPart
        if root and humanoid and humanoid.Health > 0 then
            table.insert(targets, {
                mob = mob,
                health = humanoid.Health,
                root = root
            })
        end
    end
    
    -- Sắp xếp: ít máu nhất lên đầu, nếu bằng máu thì ưu tiên gần hơn
    table.sort(targets, function(a, b)
        if a.health ~= b.health then
            return a.health < b.health
        else
            local distA = (hrp.Position - a.root.Position).Magnitude
            local distB = (hrp.Position - b.root.Position).Magnitude
            return distA < distB
        end
    end)
    
    return targets
end

    Name = "Auto Skill + Godmode V2 (Chạy Liên Tục)",
    Default = true,
    Callback = function(Value)
        AutoGodmode = Value
        
        if Value then
            mainConnection = RunService.Heartbeat:Connect(function()
                if not AutoGodmode then return end
                if not hrp or not hrp.Parent then return end
                
                local targets = GetSortedTargets()
                if #targets == 0 then return end
                
                -- Mục tiêu ưu tiên là con có máu thấp nhất
                local target = targets[1]
                local targetHRP = target.root
                local targetPos = targetHRP.Position + Vector3.new(0, HEIGHT_ABOVE, 0)
                
                -- Fix rơi + giật mạnh
                hrp.Velocity = Vector3.new(0, 0, 0)
                hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
                hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
                
                -- Tween mượt đến vị trí trên đầu mục tiêu
                local tweenInfo = TweenInfo.new(0.22, Enum.EasingStyle.Sine, Enum.EasingDirection.Out)
                local tween = TweenService:Create(hrp, tweenInfo, {
                    CFrame = CFrame.new(targetPos, targetHRP.Position)
                })
                tween:Play()
                
                -- Auto skill nếu khoảng cách <= 30
                local dist = (hrp.Position - targetHRP.Position).Magnitude
                if dist <= 30 then
                    -- Cập nhật TargetPos động cho skill (dùng ExecuteSkill với TargetPos)
